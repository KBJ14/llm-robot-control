Valid JSON Actions

Copy one of the examples below into the JSON Command box and adjust parameters as needed.

{
  "action": "send_vel",
  "parameters": { "velocity": 0.2 }
}
{
  "action": "stop",
  "parameters": {}
}
{
  "action": "toggle_auto_exploration",
  "parameters": { "resume_exploration": true }
}
{
  "action": "navigate_to_pose",
  "parameters": {
    "x": 1.5,
    "y": -2.0,
    "z_orientation": 0.0,
    "w_orientation": 1.0
  }
}
{
  "action": "navigate_relative",
  "parameters": {
    "x": 1.0,
    "y": 0.5,
    "z_orientation": 0.0,
    "w_orientation": 1.0
  }
}
{
  "action": "save_map",
  "parameters": { "map_name": "my_house_map" }
}
{
  "action": "list_saved_maps",
  "parameters": {}
}
{
  "action": "get_location_names",
  "parameters": {}
}
{
  "action": "navigate_to_location_by_name",
  "parameters": { "location_name": "kitchen" }
}

Note: the ROS bridge currently supports velocity-based commands and is being extended to cover the full set above.
